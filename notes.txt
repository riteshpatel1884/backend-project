Database
- We can set up databse in 2 ways
1. Keeping all the codes of db in index file 
In index.js
const mongoose = require("mongoose");
const {DB_NAME} = require("./constants");

const express = require("express");
const { configDotenv } = require("dotenv");
const app = express();

( async () => {
    try {
        await mongoose.connect(`${process.env.MONGODB_URL}/${DB_NAME}`);
        console.log("Connected to MongoDB");
        // production me jab database connect ho jata hai then uske baad kayi bar listener
        // dikhte hai. Listner hote hai app ke pass. So, we can add listener after database connection is successful. Menas database to connect ho gya but kya pata ki express ki app hi baat nhi kr paa rhi ho thats why we use listenr
        app.on("error", (error)=>{
            console.error("Error: ", error);
            throw error
        })

        app.listen(process.env.PORT, ()=>{
            console.log(`Server is running on port ${process.env.PORT}`);
        }
    )   

    } catch (error) {
        console.error("Error connecting to MongoDB:", error);
    }
})();
